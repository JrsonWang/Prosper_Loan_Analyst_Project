import datetime
x = datetime.datetime(2005,11,30)
 
y = datetime.datetime(2014,3,31)
 
def month_differ(x, y):
    """day, monthyear
    Parameters
    ----------
    x, y: datetime.datetime
 
    Return
    ------
    differ: x, y
    """
    month_differ = abs((x.year - y.year) * 12 + (x.month - y.month) * 1)
    return month_differ
 
print(month_differ(x, y))
